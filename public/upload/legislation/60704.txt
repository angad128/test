


public class Developer
{   
    
    protected String platform;
    protected String interviewerName;
    protected String developerName;
    protected int workingHours;
    
    public Developer(String platform, String interviewerName, int workingHours){
        
        this.platform = platform;
        this.interviewerName = interviewerName;
        this.workingHours = workingHours;
        this.developerName = "";
    }
   
    public String getPlatform(){
        return this.platform;
    }
    public String getInterviewerName(){
        return this.interviewerName;
    }
    public int getWorkingHours(){
        return this.workingHours;
    }
    public String getDeveloperName(){
        return this.developerName;
    }
        
    public void setDeveloperName(String developerName){
        this.developerName = developerName;
        
    }
    
    public void display(){
        
        System.out.println(" The platform is : " + this.getPlatform());
        System.out.println("The Working Hour is : " + this.getWorkingHours() + "hours");
        System.out.println(" The name of the Interviewer is : " + this.getInterviewerName());
        if (!(this.developerName.equals(""))){
           
            System.out.println("The name of the Developer is : " + this.getDeveloperName());
        }
    }
}
